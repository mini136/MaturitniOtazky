Tvorba BASH skriptů

Základní struktura

#!/bin/bash
# Shebang na prvním řádku říká systému, jaký interpret použít

# Spuštění:
# chmod +x skript.sh && ./skript.sh
# nebo: bash skript.sh

Proměnné

NAME="Linux"                  # bez mezer kolem =
echo "Systém: $NAME"          # interpolace v dvojitých uvozovkách
echo 'Systém: $NAME'          # jednoduché uvozovky = literální text
echo "Verze: ${NAME}_v2"      # {} pro oddělení jména proměnné

# Speciální proměnné
$0          # jméno skriptu
$1, $2...   # argumenty
$#          # počet argumentů
$@          # všechny argumenty (jako pole)
$?          # návratový kód posledního příkazu
$$          # PID aktuálního skriptu
$!          # PID posledního procesu na pozadí

# Čtení vstupu
read -p "Zadej jméno: " JMENO
echo "Ahoj, $JMENO"

# Příkaz do proměnné
DATUM=$(date +%Y-%m-%d)
SOUBORY=$(ls -1 | wc -l)

Podmínky

# if-elif-else
if [ "$1" -gt 10 ]; then
    echo "Větší než 10"
elif [ "$1" -eq 10 ]; then
    echo "Rovno 10"
else
    echo "Menší než 10"
fi

# Porovnávání čísel: -eq -ne -lt -le -gt -ge
# Porovnávání řetězců: = != -z (prázdný) -n (neprázdný)
# Testy souborů: -f (soubor) -d (adresář) -e (existuje) -r -w -x

# Modernější [[ ]]
if [[ "$STR" == *"hello"* ]]; then
    echo "Obsahuje hello"
fi

# Case
case "$1" in
    start)  echo "Starting..." ;;
    stop)   echo "Stopping..." ;;
    *)      echo "Usage: $0 {start|stop}" ;;
esac

Cykly

# For
for i in 1 2 3 4 5; do echo $i; done
for i in {1..10}; do echo $i; done
for i in $(seq 1 2 10); do echo $i; done   # krok 2
for file in *.txt; do echo "$file"; done

# C-style for
for ((i=0; i<10; i++)); do echo $i; done

# While
count=0
while [ $count -lt 5 ]; do
    echo $count
    ((count++))
done

# While read (čtení souboru řádek po řádku)
while IFS= read -r line; do
    echo "$line"
done < soubor.txt

echo "Návratový kód: $?"

Práce s textovými soubory

Základní nástroje

cat soubor.txt              # výpis celého souboru
head -n 20 soubor.txt       # prvních 20 řádků
tail -n 20 soubor.txt       # posledních 20 řádků
tail -f /var/log/syslog     # živé sledování
less soubor.txt             # stránkování
wc -l soubor.txt            # počet řádků
wc -w soubor.txt            # počet slov
sort soubor.txt             # seřazení
sort -n soubor.txt          # numerické řazení
sort -u soubor.txt          # unikátní + řazení
uniq                        # odstranění duplikátů (musí být seřazeno)
cut -d: -f1 /etc/passwd     # výběr 1. sloupce (oddělovač :)
tr 'a-z' 'A-Z' < soubor    # překlad/nahrazení znaků
paste soubor1 soubor2       # spojení souborů vedle sebe
diff soubor1 soubor2        # porovnání souborů

Přesměrování a roury

cmd > soubor        # přesměrování stdout (přepíše)
cmd >> soubor       # přesměrování stdout (přidá)
cmd 2> errors.log   # přesměrování stderr
cmd &> all.log      # stdout + stderr
cmd 2>&1            # stderr do stdout
cmd < soubor        # stdin ze souboru
cmd1 | cmd2         # roura (pipe) – stdout cmd1 → stdin cmd2

# Příklady
ps aux | grep nginx | grep -v grep
cat access.log | cut -d' ' -f1 | sort | uniq -c | sort -rn | head

Filtrace textu

grep

grep "vzor" soubor          # hledání vzoru
grep -i "vzor" soubor       # case-insensitive
grep -r "vzor" /etc/        # rekurzivní hledání
grep -n "vzor" soubor       # s čísly řádků
grep -c "vzor" soubor       # počet výskytů
grep -v "vzor" soubor       # invertovaný výběr (řádky BEZ vzoru)
grep -E "vzor1|vzor2" soubor # rozšířený regex (egrep)
grep -w "slovo" soubor      # celé slovo

sed (Stream Editor)

sed 's/starý/nový/' soubor       # nahrazení prvního výskytu na řádku
sed 's/starý/nový/g' soubor      # nahrazení všech výskytů
sed -i 's/starý/nový/g' soubor   # in-place (přepíše soubor)
sed -n '5,10p' soubor             # výpis řádků 5-10
sed '3d' soubor                   # smazání 3. řádku
sed '/vzor/d' soubor              # smazání řádků s vzorem

awk

awk '{print $1}' soubor               # 1. sloupec
awk -F: '{print $1, $3}' /etc/passwd  # oddělovač :, 1. a 3. sloupec
awk '$3 > 1000 {print $1}' /etc/passwd # podmínka
awk 'NR==5' soubor                     # 5. řádek
awk '{sum += $1} END {print sum}'      # součet 1. sloupce
awk '/vzor/ {print}' soubor            # řádky odpovídající vzoru

Kompilace

Kompilace C/C++ programu

# GCC
gcc -o program program.c              # kompilace C
g++ -o program program.cpp            # kompilace C++
gcc -Wall -O2 -o program program.c    # s varováními a optimalizací

# Kompilace s knihovnami
gcc -o program program.c -lm -lpthread

# Make
make                    # kompilace podle Makefile
make clean              # vyčištění
make install            # instalace

Makefile

CC = gcc
CFLAGS = -Wall -O2

program: main.o utils.o
	$(CC) $(CFLAGS) -o program main.o utils.o

main.o: main.c
	$(CC) $(CFLAGS) -c main.c

utils.o: utils.c utils.h
	$(CC) $(CFLAGS) -c utils.c

clean:
	rm -f *.o program

Zálohování

tar

tar -czf zaloha.tar.gz /home/user     # vytvoření komprimované zálohy (gzip)
tar -cjf zaloha.tar.bz2 /home/user   # bzip2 komprese
tar -xzf zaloha.tar.gz               # rozbalení
tar -xzf zaloha.tar.gz -C /tmp/      # rozbalení do adresáře
tar -tzf zaloha.tar.gz               # výpis obsahu

rsync

# Lokální synchronizace
rsync -av /zdrojovy/ /cilovy/

# Vzdálená záloha
rsync -avz /home/user/ user@server:/backup/
rsync -avz --delete /zdroj/ /cil/     # smazání souborů v cíli, které nejsou ve zdroji

# -a = archivní (rekurze, práva, časy...)
# -v = verbose
# -z = komprese při přenosu
# --exclude '*.tmp' = vynechání souborů

dd

# Záloha celého disku/oddílu (bit po bitu)
dd if=/dev/sda of=/backup/disk.img bs=4M status=progress
dd if=/backup/disk.img of=/dev/sdb bs=4M   # obnova

Zálohovací strategie

  • Plná záloha – kompletní kopie všech dat
  • Inkrementální – jen změny od poslední zálohy (rychlá, ale obnova složitější)
  • Diferenční – změny od poslední plné zálohy (kompromis)
  • Pravidlo 3-2-1 – 3 kopie, 2 různá média, 1 mimo lokalitu
