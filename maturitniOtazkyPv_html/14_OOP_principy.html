Ukázkový kód — Python

Zapouzdření — BankovniUcet skrývá zůstatek (__zustatek = name mangling). Přístup jen přes metody vloz(), vyber() a property zustatek.
class BankovniUcet:
    """Zapouzdreni - skryti stavu"""
    def __init__(self, majitel, zustatek=0):
        self._majitel = majitel
        self.__zustatek = zustatek   # name mangling

    @property
    def zustatek(self):
        return self.__zustatek

    def vloz(self, castka):
        if castka <= 0:
            raise ValueError("Castka musi byt kladna!")
        self.__zustatek += castka

    def vyber(self, castka):
        if castka > self.__zustatek:
            raise ValueError("Nedostatek prostredku!")
        self.__zustatek -= castka

ucet = BankovniUcet("Alice", 1000)
ucet.vloz(500)
ucet.vyber(200)
print(f"Zustatek: {ucet.zustatek} Kc")  # 1300
Polymorfismus — abstraktní třída Zivocich definuje zvuk(). Každý potomek ji implementuje jinak. Funkce udelej_zvuk() funguje s libovolným zvířetem.
from abc import ABC, abstractmethod

class Zivocich(ABC):
    def __init__(self, jmeno):
        self.jmeno = jmeno

    @abstractmethod
    def zvuk(self): pass

    def __str__(self):
        return f"{self.jmeno} rika: {self.zvuk()}"

class Pes(Zivocich):
    def zvuk(self): return "Haf haf!"

class Kocka(Zivocich):
    def zvuk(self): return "Mnau!"

class Had(Zivocich):
    def zvuk(self): return "Ssss!"

# Polymorfismus - stejna funkce, ruzne chovani
def udelej_zvuk(zivocich):
    print(f"  {zivocich}")

for z in [Pes("Rex"), Kocka("Micka"), Had("Ssisko")]:
    udelej_zvuk(z)
Agregace vs Kompozice — Motor (kompozice) je vytvořen uvnitř Auta a zanikne s ním. Řidič (agregace) existuje nezávisle na Autě.
class Motor:
    def __init__(self, typ): self.typ = typ
    def nastartuj(self): print(f"    Motor {self.typ} nastartovan")

class Ridic:
    def __init__(self, jmeno): self.jmeno = jmeno

class Auto:
    def __init__(self, znacka, motor_typ, ridic=None):
        self.znacka = znacka
        self.motor = Motor(motor_typ)   # KOMPOZICE - motor vytvoren uvnitr
        self.ridic = ridic              # AGREGACE - ridic existuje nezavisle

ridic = Ridic("Alice")
auto = Auto("Skoda", "1.6 TSI", ridic)
auto.motor.nastartuj()
print(f"  {auto.znacka} ridi {auto.ridic.jmeno}")
# Kdyz auto zanikne, motor zanikne taky (kompozice)
# Ridic ale existuje dal (agregace)
SOLID — SRP + OCP + DIP. Student má jednu zodpovědnost (data), StudentRepository druhou (ukládání). UkladaniStrategy demonstruje Dependency Inversion — ukládáme přes abstrakci.
# SRP - kazda trida jednu zodpovednost
class Student:
    def __init__(self, jmeno, vek):
        self.jmeno = jmeno
        self.vek = vek

class StudentRepository:
    """Oddelena zodpovednost: uloziste"""
    def __init__(self):
        self._studenti = []

    def pridej(self, student):
        self._studenti.append(student)

    def najdi(self, jmeno):
        return next((s for s in self._studenti if s.jmeno == jmeno), None)

# DIP - zavisej na abstrakci
class UkladaniStrategy(ABC):
    @abstractmethod
    def uloz(self, data): pass

class SouborUkladani(UkladaniStrategy):
    def uloz(self, data):
        print(f"  Ukladam do souboru: {data}")

class DBUkladani(UkladaniStrategy):
    def uloz(self, data):
        print(f"  Ukladam do DB: {data}")

class Aplikace:
    def __init__(self, ukladani: UkladaniStrategy):
        self.ukladani = ukladani   # zavisi na abstrakci, ne na implementaci

    def zpracuj(self, data):
        self.ukladani.uloz(data)

# Snadno zamenitelne
app = Aplikace(SouborUkladani())
app.zpracuj("data1")
app = Aplikace(DBUkladani())
app.zpracuj("data2")
