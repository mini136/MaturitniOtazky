Pyramida testování

ÚroveňCo testujeRychlostPokrytí
Unit testyJednotlivé funkce/metodyRychléMnoho testů
Integrační testySpolupráce komponentStředníMéně
E2E testyCelý systémPomaléMálo

AAA pattern

  1. Arrange — příprava vstupů a prostředí
  2. Act — provedení testované operace
  3. Assert — ověření výsledku

FIRST princip

  • Fast — testy běží rychle
  • Independent — nezávisí na sobě
  • Repeatable — vždy stejný výsledek
  • Self-validating — automaticky ověří pass/fail
  • Timely — píší se včas (TDD: před kódem)

TDD cyklus

  1. RED — napiš test → selže
  2. GREEN — napiš minimální kód → test projde
  3. REFACTOR — vylepši kód (testy stále zelené)

Ukázkový kód — Python

Třída Kalkulacka — testovaná třída s metodami pro sčítání, odčítání, násobení a dělení (včetně ošetření dělení nulou).
class Kalkulacka:
    """Jednoducha kalkulacka pro demonstraci testovani"""

    def secti(self, a, b):
        return a + b

    def odecti(self, a, b):
        return a - b

    def vynasob(self, a, b):
        return a * b

    def vydel(self, a, b):
        if b == 0:
            raise ZeroDivisionError("Nelze delit nulou!")
        return a / b

    def prumer(self, cisla):
        if not cisla:
            raise ValueError("Prazdny seznam!")
        return sum(cisla) / len(cisla)
Unit testy s unittest — testovací třída TestKalkulacka testuje všechny metody kalkulačky. setUp() inicializuje kalkulačku před každým testem. assertRaises ověřuje výjimky.
import unittest

class TestKalkulacka(unittest.TestCase):
    """Testy pro Kalkulacku"""

    def setUp(self):
        """Arrange - pred kazdym testem"""
        self.calc = Kalkulacka()

    def test_secti_kladna(self):
        # Arrange (hotovo v setUp)
        # Act
        vysledek = self.calc.secti(2, 3)
        # Assert
        self.assertEqual(vysledek, 5)

    def test_secti_zaporna(self):
        self.assertEqual(self.calc.secti(-1, -1), -2)

    def test_secti_nula(self):
        self.assertEqual(self.calc.secti(0, 0), 0)

    def test_odecti(self):
        self.assertEqual(self.calc.odecti(10, 3), 7)

    def test_vynasob(self):
        self.assertEqual(self.calc.vynasob(4, 5), 20)

    def test_vydel(self):
        self.assertAlmostEqual(self.calc.vydel(10, 3), 3.333, places=2)

    def test_vydel_nulou(self):
        """Test vyjimky pri deleni nulou"""
        with self.assertRaises(ZeroDivisionError):
            self.calc.vydel(10, 0)

    def test_prumer(self):
        self.assertEqual(self.calc.prumer([1, 2, 3, 4, 5]), 3.0)

    def test_prumer_prazdny(self):
        with self.assertRaises(ValueError):
            self.calc.prumer([])
Parametrizované testy — subTest() umožňuje testovat více vstupů v jednom testu. Při selhání jednoho případu se zbylé stále provedou.
class TestParametrizovane(unittest.TestCase):
    """Parametrizovane testy s subTest"""

    def test_secti_parametry(self):
        calc = Kalkulacka()
        test_cases = [
            (2, 3, 5),
            (-1, 1, 0),
            (0, 0, 0),
            (100, -50, 50),
            (0.1, 0.2, 0.3),
        ]
        for a, b, ocekavano in test_cases:
            with self.subTest(a=a, b=b):
                self.assertAlmostEqual(calc.secti(a, b), ocekavano, places=5)
Mocking — unittest.mock.Mock nahrazuje reálnou databázi falešnou. Testujeme StudentManager izolovaně od databáze pomocí mock.return_value.
from unittest.mock import Mock, patch

class Databaze:
    def najdi_studenta(self, id): pass
    def uloz_studenta(self, student): pass

class StudentManager:
    def __init__(self, db):
        self.db = db

    def get_student_info(self, id):
        student = self.db.najdi_studenta(id)
        if student is None:
            raise ValueError(f"Student {id} nenalezen")
        return f"{student['jmeno']} (vek: {student['vek']})"

class TestStudentManagerMock(unittest.TestCase):
    def test_get_info(self):
        # Arrange - mock databaze
        mock_db = Mock(spec=Databaze)
        mock_db.najdi_studenta.return_value = {"jmeno": "Alice", "vek": 20}

        manager = StudentManager(mock_db)

        # Act
        info = manager.get_student_info(1)

        # Assert
        self.assertEqual(info, "Alice (vek: 20)")
        mock_db.najdi_studenta.assert_called_once_with(1)

    def test_student_nenalezen(self):
        mock_db = Mock(spec=Databaze)
        mock_db.najdi_studenta.return_value = None

        manager = StudentManager(mock_db)

        with self.assertRaises(ValueError):
            manager.get_student_info(999)
