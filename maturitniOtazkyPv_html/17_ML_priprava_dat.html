Ukázkový kód — Python

Syntetický dataset — vytvoření náhodných dat studentů s chybějícími hodnotami (None) pro demonstraci čištění dat.
import random
import math

random.seed(42)

# 1. Synteticky dataset
def generuj_studenty(n=100):
    jmena = ["Alice", "Bob", "Charlie", "Diana", "Eva",
             "Filip", "Gita", "Hana", "Ivan", "Jana"]
    data = []
    for i in range(n):
        student = {
            "id": i+1,
            "jmeno": random.choice(jmena),
            "vek": random.randint(18, 30),
            "hodiny_studia": round(random.uniform(0, 40), 1),
            "predchozi_znamka": round(random.uniform(1, 5), 1),
            "znamka": None  # target
        }
        # Simulace vztahu: vice studia = lepsi znamka
        student["znamka"] = round(
            5 - (student["hodiny_studia"]/40) * 3
            + random.uniform(-0.5, 0.5), 1)
        student["znamka"] = max(1, min(5, student["znamka"]))

        # Simulace chybejicich hodnot (~10%)
        if random.random() < 0.1:
            student["hodiny_studia"] = None
        if random.random() < 0.05:
            student["vek"] = None
        data.append(student)
    return data

studenti = generuj_studenty(100)
print(f"Pocet studentu: {len(studenti)}")
print(f"Prvni: {studenti[0]}")
Imputace chybějících hodnot — nahrazení None průměrem (mean), mediánem nebo nejčastější hodnotou (modem). Funkce imputuj() detekuje a nahradí chybějící data.
# 2. Imputace - nahrazeni chybejicich hodnot
def imputuj(data, sloupec, metoda="mean"):
    """Nahradi None hodnotou podle zvolene metody"""
    hodnoty = [d[sloupec] for d in data if d[sloupec] is not None]
    chybejici = sum(1 for d in data if d[sloupec] is None)

    if metoda == "mean":
        nahrada = sum(hodnoty) / len(hodnoty)
    elif metoda == "median":
        s = sorted(hodnoty)
        n = len(s)
        nahrada = s[n//2] if n % 2 else (s[n//2-1] + s[n//2]) / 2
    elif metoda == "mode":
        nahrada = max(set(hodnoty), key=hodnoty.count)

    for d in data:
        if d[sloupec] is None:
            d[sloupec] = round(nahrada, 1)

    print(f"  {sloupec}: {chybejici} chybejicich → nahrazeno ({metoda}: {nahrada:.1f})")

imputuj(studenti, "hodiny_studia", "mean")
imputuj(studenti, "vek", "median")
Detekce outlierů pomocí IQR (interkvartilový rozsah). Hodnoty mimo [Q1 − 1.5×IQR, Q3 + 1.5×IQR] jsou považovány za odlehlé.
# 3. Detekce outlieru (IQR metoda)
def detekuj_outliery(data, sloupec):
    hodnoty = sorted(d[sloupec] for d in data)
    n = len(hodnoty)
    q1 = hodnoty[n // 4]
    q3 = hodnoty[3 * n // 4]
    iqr = q3 - q1
    dolni = q1 - 1.5 * iqr
    horni = q3 + 1.5 * iqr

    outliery = [d for d in data if d[sloupec] < dolni or d[sloupec] > horni]
    print(f"  {sloupec}: IQR={iqr:.1f}, rozsah=[{dolni:.1f}, {horni:.1f}], outlieru: {len(outliery)}")
    return outliery

detekuj_outliery(studenti, "hodiny_studia")
Min-Max a Z-score normalizace — přeškálování hodnot do rozsahu [0,1] resp. standardizace na střed 0 a směrodatnou odchylku 1.
# 4. Normalizace
def min_max(data, sloupec):
    hodnoty = [d[sloupec] for d in data]
    min_v, max_v = min(hodnoty), max(hodnoty)
    for d in data:
        d[f"{sloupec}_norm"] = round((d[sloupec] - min_v) / (max_v - min_v), 3)

def z_score(data, sloupec):
    hodnoty = [d[sloupec] for d in data]
    prumer = sum(hodnoty) / len(hodnoty)
    odchylka = math.sqrt(sum((x - prumer)**2 for x in hodnoty) / len(hodnoty))
    for d in data:
        d[f"{sloupec}_zscore"] = round((d[sloupec] - prumer) / odchylka, 3) if odchylka else 0

min_max(studenti, "hodiny_studia")
z_score(studenti, "predchozi_znamka")
print(f"  Ukazka: norm={studenti[0].get('hodiny_studia_norm')}, "
      f"z={studenti[0].get('predchozi_znamka_zscore')}")
One-hot encoding — převod kategorické proměnné (jméno) na binární sloupce. Každá kategorie dostane vlastní sloupec (0/1).
# 5. One-hot encoding
def one_hot(data, sloupec):
    unikatni = sorted(set(d[sloupec] for d in data))
    print(f"  One-hot '{sloupec}': {len(unikatni)} kategorii")
    for d in data:
        for kat in unikatni:
            d[f"{sloupec}_{kat}"] = 1 if d[sloupec] == kat else 0

one_hot(studenti, "jmeno")
print(f"  Ukazka sloupcu: {[k for k in studenti[0] if k.startswith('jmeno_')]}")
Korelace — Pearsonův korelační koeficient měří lineární závislost dvou proměnných. Hodnota blízká −1 nebo +1 znamená silnou korelaci.
# 6. Korelace
def korelace(data, sl1, sl2):
    x = [d[sl1] for d in data]
    y = [d[sl2] for d in data]
    n = len(x)
    mx, my = sum(x)/n, sum(y)/n
    cov = sum((x[i]-mx)*(y[i]-my) for i in range(n)) / n
    sx = math.sqrt(sum((xi-mx)**2 for xi in x) / n)
    sy = math.sqrt(sum((yi-my)**2 for yi in y) / n)
    r = cov / (sx * sy) if sx and sy else 0
    print(f"  Korelace({sl1}, {sl2}) = {r:.3f}")
    return r

korelace(studenti, "hodiny_studia", "znamka")
korelace(studenti, "predchozi_znamka", "znamka")
