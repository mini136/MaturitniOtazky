Ukázkový kód — Python

Dynamické typování — proměnná x mění typ za běhu. type() vrací aktuální typ. Python je silně typovaný (nelze sčítat int + str bez konverze).
# Dynamicke typovani - typ se urcuje za behu
x = 42
print(f"x = {x}, typ: {type(x)}")  # <class 'int'>

x = "ahoj"
print(f"x = {x}, typ: {type(x)}")  # <class 'str'>

x = [1, 2, 3]
print(f"x = {x}, typ: {type(x)}")  # <class 'list'>
Duck typing — "If it walks like a duck..." Funkce spust() volá start() na čemkoli, co tuto metodu má — nezáleží na typu.
class Auto:
    def start(self): return "Motor nastartovan"

class Pocitac:
    def start(self): return "System spusten"

class Mikrovlnka:
    def start(self): return "Ohrivam..."

def spust(vec):
    """Nezalezi na typu - staci mit metodu start()"""
    print(f"  {type(vec).__name__}: {vec.start()}")

for obj in [Auto(), Pocitac(), Mikrovlnka()]:
    spust(obj)
Generátory — líné vyhodnocování pomocí yield. Generují hodnoty jen na požádání, šetří paměť. fibonacci() produkuje nekonečnou sekvenci.
def fibonacci():
    """Generator - line vyhodnocovani"""
    a, b = 0, 1
    while True:
        yield a
        a, b = b, a + b

# Generujeme jen prvnich 10
fib = fibonacci()
prvnich_10 = [next(fib) for _ in range(10)]
print(f"Fibonacci: {prvnich_10}")

# Generator expression vs list comprehension
gen = (x**2 for x in range(1000000))  # ZADNA pamet
lst = [x**2 for x in range(10)]       # cele v pameti
print(f"Generator: {type(gen)}")
print(f"List:      {type(lst)}")
Dekorátory — funkce, která obaluje jinou funkci a přidává chování (logování, měření času). @timer automaticky měří dobu běhu funkce.
import time

def timer(func):
    """Dekorator - meri cas"""
    def wrapper(*args, **kwargs):
        start = time.time()
        vysledek = func(*args, **kwargs)
        konec = time.time()
        print(f"  {func.__name__}: {konec - start:.4f}s")
        return vysledek
    return wrapper

@timer
def pomala_funkce():
    total = sum(range(1_000_000))
    return total

vysledek = pomala_funkce()
Bytecode — modul dis ukazuje, jak CPython interně kompiluje Python kód do bytecodu před interpretací.
import dis

def secti(a, b):
    return a + b

print("Bytecode funkce secti():")
dis.dis(secti)
# LOAD_FAST 0 (a)
# LOAD_FAST 1 (b)
# BINARY_ADD
# RETURN_VALUE

Ukázkový kód — Java

Java — statické typování, generika, String pool. Java vyžaduje deklaraci typů při kompilaci. String.intern() demonstruje String pool v JVM.
public class JavaUkazka {
    // Staticke typovani - typ se urcuje pri kompilaci
    public static void main(String[] args) {
        int cislo = 42;                        // primitivni typ
        String text = "Ahoj";                  // referencni typ
        double[] pole = {1.1, 2.2, 3.3};       // pole

        // Generika - typova bezpecnost
        java.util.List<String> jmena = new java.util.ArrayList<>();
        jmena.add("Alice");
        jmena.add("Bob");
        // jmena.add(42);  // CHYBA pri kompilaci!

        // String pool
        String s1 = "hello";
        String s2 = "hello";
        String s3 = new String("hello");
        System.out.println(s1 == s2);              // true  (stejny objekt z poolu)
        System.out.println(s1 == s3);              // false (novy objekt na heapu)
        System.out.println(s1.equals(s3));         // true  (porovnani obsahu)
    }
}
