Lineární regrese — implementace gradientního sestupu. Iterativně optimalizuje parametry w (váha) a b (bias) minimalizací MSE.
import math
import random

random.seed(42)

class LinearniRegrese:
    """Linearni regrese s gradientnim sestupem"""

    def __init__(self, lr=0.01, iterace=1000):
        self.lr = lr
        self.iterace = iterace
        self.w = 0
        self.b = 0

    def fit(self, X, y):
        n = len(X)
        for i in range(self.iterace):
            # Predikce
            y_pred = [self.w * x + self.b for x in X]

            # Gradienty
            dw = (-2/n) * sum((y[j] - y_pred[j]) * X[j] for j in range(n))
            db = (-2/n) * sum(y[j] - y_pred[j] for j in range(n))

            # Aktualizace
            self.w -= self.lr * dw
            self.b -= self.lr * db

            if i % 200 == 0:
                mse = sum((y[j] - y_pred[j])**2 for j in range(n)) / n
                print(f"    Iterace {i}: MSE={mse:.4f}")

    def predict(self, X):
        return [self.w * x + self.b for x in X]

# Data: hodiny studia → znamka
X = [5, 10, 15, 20, 25, 30, 35]
y = [4.5, 3.8, 3.2, 2.5, 2.0, 1.5, 1.2]

model = LinearniRegrese(lr=0.001, iterace=1000)
model.fit(X, y)
print(f"  w={model.w:.3f}, b={model.b:.3f}")
print(f"  Predikce pro 22h: {model.predict([22])[0]:.2f}")
Logistická regrese — binární klasifikace pomocí sigmoidní funkce. Sigmoid mapuje libovolnou hodnotu do rozsahu (0, 1) jako pravděpodobnost.
class LogistickaRegrese:
    """Logisticka regrese - binarni klasifikace"""

    def __init__(self, lr=0.1, iterace=1000):
        self.lr = lr
        self.iterace = iterace
        self.w = [0, 0]
        self.b = 0

    def sigmoid(self, z):
        z = max(-500, min(500, z))
        return 1 / (1 + math.exp(-z))

    def fit(self, X, y):
        n = len(X)
        for i in range(self.iterace):
            for j in range(n):
                z = sum(self.w[k] * X[j][k] for k in range(len(self.w))) + self.b
                pred = self.sigmoid(z)
                chyba = pred - y[j]
                for k in range(len(self.w)):
                    self.w[k] -= self.lr * chyba * X[j][k]
                self.b -= self.lr * chyba

    def predict(self, X):
        vysledky = []
        for x in X:
            z = sum(self.w[k] * x[k] for k in range(len(self.w))) + self.b
            vysledky.append(1 if self.sigmoid(z) >= 0.5 else 0)
        return vysledky

# Data: [hodiny_studia, predchozi_znamka] → prosli (1) / neprosli (0)
X_log = [[30, 1.5], [25, 2.0], [20, 2.5], [35, 1.2],
         [5, 4.5], [10, 3.8], [8, 4.0], [15, 3.5]]
y_log = [1, 1, 1, 1, 0, 0, 0, 0]

log_model = LogistickaRegrese(lr=0.01, iterace=500)
log_model.fit(X_log, y_log)
predikce = log_model.predict([[22, 2.0], [7, 4.2]])
print(f"  Predikce: {predikce}")  # [1, 0]
KNN klasifikátor — klasifikuje ovoce podle váhy a sladkosti. Najde K nejbližších sousedů (euklidovská vzdálenost) a vrátí nejčastější třídu.
class KNN:
    """K-Nearest Neighbors"""

    def __init__(self, k=3):
        self.k = k
        self.X_train = []
        self.y_train = []

    def fit(self, X, y):
        self.X_train = X
        self.y_train = y

    def vzdalenost(self, a, b):
        return math.sqrt(sum((a[i]-b[i])**2 for i in range(len(a))))

    def predict_one(self, x):
        # Vzdalenosti ke vsem treninkovym bodum
        vzdalenosti = [(self.vzdalenost(x, self.X_train[i]), self.y_train[i])
                       for i in range(len(self.X_train))]
        vzdalenosti.sort(key=lambda v: v[0])

        # K nejblizsich sousedu
        k_sousedi = [v[1] for v in vzdalenosti[:self.k]]
        # Nejcastejsi trida
        return max(set(k_sousedi), key=k_sousedi.count)

    def predict(self, X):
        return [self.predict_one(x) for x in X]

# Data: [vaha_g, sladkost] → ovoce
X_ovoce = [[150, 7], [170, 8], [140, 6],   # jablko
            [80, 3], [90, 4], [70, 2],       # citron
            [200, 9], [180, 10], [220, 8]]   # mango
y_ovoce = ["jablko"]*3 + ["citron"]*3 + ["mango"]*3

knn = KNN(k=3)
knn.fit(X_ovoce, y_ovoce)
print(f"  KNN [160,7]: {knn.predict_one([160, 7])}")  # jablko
print(f"  KNN [85,3]:  {knn.predict_one([85, 3])}")   # citron
Confusion Matrix — výpočet TP, TN, FP, FN a odvozených metrik (accuracy, precision, recall, F1) z predikovaných a skutečných hodnot.
def confusion_matrix(y_true, y_pred):
    """Matice zameny + metriky"""
    tp = sum(1 for t, p in zip(y_true, y_pred) if t == 1 and p == 1)
    tn = sum(1 for t, p in zip(y_true, y_pred) if t == 0 and p == 0)
    fp = sum(1 for t, p in zip(y_true, y_pred) if t == 0 and p == 1)
    fn = sum(1 for t, p in zip(y_true, y_pred) if t == 1 and p == 0)

    accuracy  = (tp + tn) / (tp + tn + fp + fn) if (tp+tn+fp+fn) else 0
    precision = tp / (tp + fp) if (tp + fp) else 0
    recall    = tp / (tp + fn) if (tp + fn) else 0
    f1 = 2*precision*recall / (precision+recall) if (precision+recall) else 0

    print(f"  TP={tp}, TN={tn}, FP={fp}, FN={fn}")
    print(f"  Accuracy:  {accuracy:.3f}")
    print(f"  Precision: {precision:.3f}")
    print(f"  Recall:    {recall:.3f}")
    print(f"  F1 Score:  {f1:.3f}")

y_true = [1, 1, 0, 1, 0, 0, 1, 0, 1, 0]
y_pred = [1, 1, 0, 0, 0, 1, 1, 0, 1, 0]
confusion_matrix(y_true, y_pred)
