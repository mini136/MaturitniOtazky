Ukázkový kód — Python

Základní try-except-else-finally — kompletní ukázka: else se provede jen bez výjimky, finally vždy.
def bezpecne_deleni(a, b):
    try:
        vysledek = a / b
    except ZeroDivisionError:
        print("  Deleni nulou!")
        return None
    except TypeError as e:
        print(f"  Spatny typ: {e}")
        return None
    else:
        print(f"  Vysledek: {vysledek}")
        return vysledek
    finally:
        print("  Finally blok (vzdy se provede)")

bezpecne_deleni(10, 3)
bezpecne_deleni(10, 0)
bezpecne_deleni("a", 2)
Vlastní výjimky — hierarchie validačních chyb: ValidacniChyba jako základ, VekChyba a JmenoChyba jako specifické podtřídy.
class ValidacniChyba(Exception):
    """Zakladni validacni chyba"""
    def __init__(self, zprava, pole=None):
        super().__init__(zprava)
        self.pole = pole

class VekChyba(ValidacniChyba):
    def __init__(self, vek):
        super().__init__(f"Neplatny vek: {vek}", pole="vek")
        self.vek = vek

class JmenoChyba(ValidacniChyba):
    def __init__(self, jmeno):
        super().__init__(f"Neplatne jmeno: '{jmeno}'", pole="jmeno")

def validuj_studenta(jmeno, vek):
    if not jmeno or len(jmeno) < 2:
        raise JmenoChyba(jmeno)
    if not isinstance(vek, int) or vek < 15 or vek > 99:
        raise VekChyba(vek)
    return {"jmeno": jmeno, "vek": vek}

# Zachyceni
for jmeno, vek in [("Alice", 20), ("", 20), ("Bob", 10)]:
    try:
        student = validuj_studenta(jmeno, vek)
        print(f"  OK: {student}")
    except JmenoChyba as e:
        print(f"  Chyba jmena: {e} (pole: {e.pole})")
    except VekChyba as e:
        print(f"  Chyba veku: {e} (vek: {e.vek})")
Exception chaining — raise ... from e zachovává původní výjimku jako příčinu. Umožňuje trasovat řetěz chyb při debugování.
def nacti_konfiguraci(cesta):
    try:
        with open(cesta) as f:
            return f.read()
    except FileNotFoundError as e:
        raise RuntimeError(f"Konfigurace '{cesta}' chybi!") from e

try:
    nacti_konfiguraci("neexistuje.conf")
except RuntimeError as e:
    print(f"  Chyba: {e}")
    print(f"  Pricina: {e.__cause__}")
Assertions — preconditions (vstupní podmínky), postconditions (výstupní podmínky) a invarianty (podmínky platné po celou dobu). Lze vypnout s python -O.
def faktorial(n):
    # Precondition
    assert isinstance(n, int) and n >= 0, f"n musi byt nezaporne cele cislo, dostali jsme {n}"

    vysledek = 1
    for i in range(2, n + 1):
        vysledek *= i

    # Postcondition
    assert vysledek > 0, "Faktorial musi byt kladny"
    return vysledek

print(f"  5! = {faktorial(5)}")   # 120

# Invariant v datove strukture
class SerazenyBuffer:
    def __init__(self):
        self._data = []

    def vloz(self, hodnota):
        self._data.append(hodnota)
        self._data.sort()
        # Invariant: data musi byt serazena
        assert all(self._data[i] <= self._data[i+1]
                   for i in range(len(self._data)-1)), "Buffer neni serazeny!"

    def __repr__(self):
        return str(self._data)

buf = SerazenyBuffer()
for x in [5, 2, 8, 1]:
    buf.vloz(x)
print(f"  Buffer: {buf}")  # [1, 2, 5, 8]
Logging — konfigurace loggeru s formatováním, úrovněmi a výstupem do konzole. Každý level má jiný účel (DEBUG pro vývoj, ERROR pro chyby).
import logging

logging.basicConfig(
    level=logging.DEBUG,
    format='%(asctime)s [%(levelname)s] %(name)s: %(message)s',
    datefmt='%H:%M:%S'
)
logger = logging.getLogger("MojeApp")

logger.debug("Detailni info pro vyvoj")
logger.info("Bezna informace")
logger.warning("Varovani - neco by mohlo byt spatne")
logger.error("Chyba - neco selhalo")
logger.critical("Kriticka chyba - aplikace konci")
Context manager pro cleanup — třída Databaze implementuje __enter__/__exit__. Spojení se uzavře i při výjimce.
class Databaze:
    """Context manager pro automaticky cleanup"""
    def __init__(self, nazev):
        self.nazev = nazev
        self.pripojeno = False

    def __enter__(self):
        print(f"  Pripojuji k DB '{self.nazev}'...")
        self.pripojeno = True
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):
        print(f"  Odpojuji od DB '{self.nazev}'")
        self.pripojeno = False
        if exc_type:
            print(f"  Chyba: {exc_val}")
        return False  # nepotlacuje vyjimku

# Pouziti
with Databaze("studenti_db") as db:
    print(f"  Pripojeno: {db.pripojeno}")
    # db operace...
Debugging tipy — f-string s = pro rychlý výpis proměnných, traceback modul pro detailní výpis zásobníku volání.
import traceback

# F-string debugging (Python 3.8+)
x = 42
y = "hello"
print(f"  {x=}, {y=}")        # x=42, y='hello'

# Traceback
try:
    1 / 0
except:
    print("  Traceback:")
    traceback.print_exc()
