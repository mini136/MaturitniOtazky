Ukázkový kód — Python (SQLite CRUD, transakce, ORM vzor)

CRUD operace s SQLite — vytvoření tabulky, vložení, čtení, aktualizace a mazání záznamů. Používá parametrizované dotazy (?).
import sqlite3

# In-memory databaze
conn = sqlite3.connect(':memory:')
conn.row_factory = sqlite3.Row  # pristup pres nazvy sloupcu
cursor = conn.cursor()

# CREATE tabulky
cursor.execute('''
    CREATE TABLE studenti (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        jmeno TEXT NOT NULL,
        vek INTEGER CHECK(vek > 0),
        prumer REAL DEFAULT 0.0
    )
''')

# INSERT (Create)
studenti_data = [("Alice", 20, 1.5), ("Bob", 22, 2.1), ("Charlie", 21, 1.8)]
cursor.executemany(
    "INSERT INTO studenti (jmeno, vek, prumer) VALUES (?, ?, ?)",
    studenti_data
)
conn.commit()

# SELECT (Read)
cursor.execute("SELECT * FROM studenti WHERE vek >= ?", (21,))
for row in cursor.fetchall():
    print(f"  {dict(row)}")

# UPDATE
cursor.execute("UPDATE studenti SET prumer = ? WHERE jmeno = ?", (1.2, "Alice"))

# DELETE
cursor.execute("DELETE FROM studenti WHERE jmeno = ?", ("Charlie",))
conn.commit()
Transakce — atomické operace s try/commit/rollback. Pokud kterýkoli krok selže, celá transakce se vrátí zpět.
def prevod_kreditu(conn, od, komu, castka):
    """Atomicky prevod kreditu mezi studenty"""
    try:
        cursor = conn.cursor()
        # Kontrola zustatku
        cursor.execute("SELECT prumer FROM studenti WHERE jmeno = ?", (od,))
        row = cursor.fetchone()
        if row is None or row['prumer'] < castka:
            raise ValueError("Nedostatek kreditu!")

        cursor.execute("UPDATE studenti SET prumer = prumer - ? WHERE jmeno = ?",
                       (castka, od))
        cursor.execute("UPDATE studenti SET prumer = prumer + ? WHERE jmeno = ?",
                       (castka, komu))
        conn.commit()
        print(f"  Prevod {castka} od {od} ke {komu}: OK")
    except Exception as e:
        conn.rollback()
        print(f"  Prevod SELHAL: {e}")

prevod_kreditu(conn, "Alice", "Bob", 0.5)
ORM vzor — StudentRepository zapouzdřuje přímé SQL dotazy. Třída Student (dataclass) reprezentuje řádek tabulky.
from dataclasses import dataclass

@dataclass
class Student:
    id: int = 0
    jmeno: str = ""
    vek: int = 0
    prumer: float = 0.0

class StudentRepository:
    """Jednoduchy ORM-like pattern"""

    def __init__(self, conn):
        self.conn = conn

    def najdi_vsechny(self):
        cursor = self.conn.execute("SELECT * FROM studenti")
        return [Student(**dict(row)) for row in cursor.fetchall()]

    def najdi_podle_id(self, id):
        cursor = self.conn.execute("SELECT * FROM studenti WHERE id = ?", (id,))
        row = cursor.fetchone()
        return Student(**dict(row)) if row else None

    def uloz(self, student):
        if student.id == 0:
            self.conn.execute(
                "INSERT INTO studenti (jmeno, vek, prumer) VALUES (?, ?, ?)",
                (student.jmeno, student.vek, student.prumer))
        else:
            self.conn.execute(
                "UPDATE studenti SET jmeno=?, vek=?, prumer=? WHERE id=?",
                (student.jmeno, student.vek, student.prumer, student.id))
        self.conn.commit()

    def smaz(self, id):
        self.conn.execute("DELETE FROM studenti WHERE id = ?", (id,))
        self.conn.commit()

# Pouziti
repo = StudentRepository(conn)
for s in repo.najdi_vsechny():
    print(f"  {s}")

conn.close()
