Benchmark — Array vs Linked List. Porovnává výkon přístupu podle indexu (array vyhrává) a vkládání na začátek (linked list vyhrává).
import time

class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def pridej_zacatek(self, data):
        novy = Node(data)
        novy.next = self.head
        self.head = novy

# Benchmark: vlozeni na zacatek
n = 10000
start = time.perf_counter()
pole = []
for i in range(n):
    pole.insert(0, i)           # O(n) — posuny
cas_pole = time.perf_counter() - start

start = time.perf_counter()
ll = LinkedList()
for i in range(n):
    ll.pridej_zacatek(i)        # O(1) — zmena pointeru
cas_ll = time.perf_counter() - start

print(f"  Vlozeni na zacatek ({n}x):")
print(f"    Array:      {cas_pole:.4f}s")
print(f"    LinkedList: {cas_ll:.4f}s")
Hash funkce — různé metody hashování (dělení, násobení, FNV-1a). Každá metoda ukazuje jiný přístup k distribuci klíčů do bucketů.
def hash_deleni(klic, velikost):
    """Metoda deleni: h(k) = k mod m"""
    if isinstance(klic, str):
        klic = sum(ord(c) for c in klic)
    return klic % velikost

def hash_nasobeni(klic, velikost):
    """Metoda nasobeni: h(k) = floor(m * (k*A mod 1))"""
    import math
    A = 0.6180339887  # (sqrt(5) - 1) / 2
    if isinstance(klic, str):
        klic = sum(ord(c) * (31 ** i) for i, c in enumerate(klic))
    return int(velikost * ((klic * A) % 1))

def hash_fnv1a(klic, velikost):
    """FNV-1a hash"""
    h = 2166136261
    for byte in klic.encode():
        h ^= byte
        h = (h * 16777619) & 0xFFFFFFFF
    return h % velikost

# Test distribuce
for klic in ["alice", "bob", "charlie"]:
    print(f"  '{klic}':")
    print(f"    deleni:   {hash_deleni(klic, 10)}")
    print(f"    nasobeni: {hash_nasobeni(klic, 10)}")
    print(f"    FNV-1a:   {hash_fnv1a(klic, 10)}")
Hash tabulka — Chaining vs Open Addressing. Dvě implementace řešení kolizí: chaining (linked list v bucketu) a open addressing (linear probing).
class HashChaining:
    def __init__(self, vel=7):
        self.vel = vel
        self.tabulka = [[] for _ in range(vel)]
        self.pocet = 0

    def _hash(self, klic):
        return hash(klic) % self.vel

    def vloz(self, klic, hodnota):
        idx = self._hash(klic)
        for par in self.tabulka[idx]:
            if par[0] == klic:
                par[1] = hodnota
                return
        self.tabulka[idx].append([klic, hodnota])
        self.pocet += 1

    def load_factor(self):
        return self.pocet / self.vel

class HashOpenAddressing:
    def __init__(self, vel=7):
        self.vel = vel
        self.tabulka = [None] * vel
        self.pocet = 0

    def _hash(self, klic):
        return hash(klic) % self.vel

    def vloz(self, klic, hodnota):
        idx = self._hash(klic)
        for i in range(self.vel):
            pos = (idx + i) % self.vel     # linear probing
            if self.tabulka[pos] is None or self.tabulka[pos][0] == klic:
                self.tabulka[pos] = (klic, hodnota)
                self.pocet += 1
                return
        raise OverflowError("Tabulka plna!")

# Porovnani
ch = HashChaining(7)
oa = HashOpenAddressing(7)
for k, v in [("a", 1), ("b", 2), ("c", 3), ("d", 4)]:
    ch.vloz(k, v)
    oa.vloz(k, v)

print(f"  Chaining LF: {ch.load_factor():.2f}")
print(f"  Chaining:    {[b for b in ch.tabulka if b]}")
print(f"  OpenAddr:    {[x for x in oa.tabulka if x]}")
Binární vyhledávání — funguje jen na seřazených datech. Složitost O(log n) — v každém kroku zmenší prohledávaný rozsah na polovinu.
def binarni_hledani(pole, hledany):
    """O(log n) - vyzaduje serazene pole"""
    levy, pravy = 0, len(pole) - 1
    kroky = 0
    while levy <= pravy:
        kroky += 1
        stred = (levy + pravy) // 2
        if pole[stred] == hledany:
            return stred, kroky
        elif pole[stred] < hledany:
            levy = stred + 1
        else:
            pravy = stred - 1
    return -1, kroky

data = list(range(0, 1000, 2))  # [0, 2, 4, ..., 998]
idx, kroky = binarni_hledani(data, 742)
print(f"  Hledam 742 v {len(data)} prvcich: index={idx}, kroky={kroky}")
