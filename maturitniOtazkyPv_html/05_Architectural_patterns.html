Klient-Server, P2P

  • Klient-Server — klient (požadavky) ↔ server (odpovědi), centralizovaná správa
  • P2P (Peer-to-Peer) — každý uzel je klient i server (BitTorrent, blockchain)

Ukázkový kód — MVC (Python)

Model — datová vrstva spravuje seznam studentů. Obsahuje logiku pro přidání, odebrání a získání dat.
# === MODEL ===
class StudentModel:
    """Datova vrstva - spravuje data"""
    def __init__(self):
        self._studenti = []

    def pridej(self, jmeno, vek):
        student = {"id": len(self._studenti) + 1, "jmeno": jmeno, "vek": vek}
        self._studenti.append(student)
        return student

    def vsichni(self):
        return list(self._studenti)

    def odeber(self, id):
        self._studenti = [s for s in self._studenti if s["id"] != id]
View — prezentační vrstva. Pouze zobrazuje data, nemá logiku. Controller — zprostředkovává mezi modelem a view.
# === VIEW ===
class StudentView:
    """Prezentacni vrstva - zobrazuje data"""
    def zobraz_studenty(self, studenti):
        print("  Seznam studentu:")
        for s in studenti:
            print(f"    [{s['id']}] {s['jmeno']}, {s['vek']} let")

    def zobraz_zpravu(self, zprava):
        print(f"  > {zprava}")

# === CONTROLLER ===
class StudentController:
    """Ridici vrstva - zprostredkovava Model a View"""
    def __init__(self, model, view):
        self.model = model
        self.view = view

    def pridej_studenta(self, jmeno, vek):
        student = self.model.pridej(jmeno, vek)
        self.view.zobraz_zpravu(f"Pridan: {student['jmeno']}")

    def zobraz_vsechny(self):
        studenti = self.model.vsichni()
        self.view.zobraz_studenty(studenti)

# Pouziti MVC
model = StudentModel()
view = StudentView()
controller = StudentController(model, view)
controller.pridej_studenta("Alice", 20)
controller.pridej_studenta("Bob", 22)
controller.zobraz_vsechny()

Ukázkový kód — TCP klient-server

TCP server s vlákny — každé připojení obsluhuje samostatné vlákno. Přijímá data a odesílá je zpět velkými písmeny (echo server).
import socket
import threading

def tcp_server():
    """Jednoduchy TCP echo server s vlakny"""
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    server.bind(('localhost', 65432))
    server.listen(5)
    print("  Server posloucha na localhost:65432")

    def obsluha_klienta(conn, addr):
        print(f"  Pripojeni od {addr}")
        while True:
            data = conn.recv(1024)
            if not data:
                break
            odpoved = data.decode().upper()
            conn.sendall(odpoved.encode())
        conn.close()
        print(f"  Odpojeni {addr}")

    while True:
        conn, addr = server.accept()
        vlakno = threading.Thread(target=obsluha_klienta, args=(conn, addr))
        vlakno.daemon = True
        vlakno.start()
TCP klient — připojí se k serveru, odešle zprávu, přijme odpověď a uzavře spojení.
def tcp_klient():
    """TCP klient"""
    klient = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    klient.connect(('localhost', 65432))
    klient.sendall(b"Ahoj servere!")
    odpoved = klient.recv(1024)
    print(f"  Odpoved: {odpoved.decode()}")
    klient.close()
