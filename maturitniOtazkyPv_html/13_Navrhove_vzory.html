Singleton — zajišťuje jedinou instanci třídy v celé aplikaci. Zde implementováno přes __new__ s _instance atributem.
class Konfigurace:
    """Singleton - jedina instance"""
    _instance = None

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance.nastaveni = {}
        return cls._instance

    def nastav(self, klic, hodnota):
        self.nastaveni[klic] = hodnota

    def get(self, klic, default=None):
        return self.nastaveni.get(klic, default)

# Obe promenne ukazuji na STEJNOU instanci
k1 = Konfigurace()
k2 = Konfigurace()
k1.nastav("debug", True)
print(f"k1 is k2: {k1 is k2}")           # True
print(f"k2.debug: {k2.get('debug')}")     # True
Factory — vytváří objekty bez specifikace přesné třídy. Metoda vytvor() podle typu vrátí správnou instanci.
from abc import ABC, abstractmethod

class Notifikace(ABC):
    @abstractmethod
    def odesli(self, zprava): pass

class EmailNotifikace(Notifikace):
    def odesli(self, zprava):
        print(f"  EMAIL: {zprava}")

class SMSNotifikace(Notifikace):
    def odesli(self, zprava):
        print(f"  SMS: {zprava}")

class PushNotifikace(Notifikace):
    def odesli(self, zprava):
        print(f"  PUSH: {zprava}")

class NotifikaceFactory:
    """Factory - vytvari objekty podle typu"""
    _typy = {
        "email": EmailNotifikace,
        "sms": SMSNotifikace,
        "push": PushNotifikace
    }

    @classmethod
    def vytvor(cls, typ):
        trida = cls._typy.get(typ)
        if trida is None:
            raise ValueError(f"Neznamy typ: {typ}")
        return trida()

# Pouziti - klient nezna konkretni tridy
for typ in ["email", "sms", "push"]:
    notifikace = NotifikaceFactory.vytvor(typ)
    notifikace.odesli(f"Zprava pres {typ}")
Builder — krok za krokem vytváří složitý objekt. Umožňuje řetězení volání (return self).
class PizzaBuilder:
    def __init__(self):
        self.velikost = "stredni"
        self.testo = "klasicke"
        self.ingredience = []

    def nastav_velikost(self, v):
        self.velikost = v
        return self  # retezeni

    def nastav_testo(self, t):
        self.testo = t
        return self

    def pridej(self, ingredience):
        self.ingredience.append(ingredience)
        return self

    def postav(self):
        return {
            "velikost": self.velikost,
            "testo": self.testo,
            "ingredience": self.ingredience
        }

# Retezeni volani (fluent interface)
pizza = (PizzaBuilder()
    .nastav_velikost("velka")
    .nastav_testo("tenke")
    .pridej("syr")
    .pridej("sunka")
    .pridej("olivy")
    .postav())
print(f"Pizza: {pizza}")
Observer — publisher oznamuje změny všem registrovaným subscriberům (event systém).
class EventSystem:
    """Observer - publisher / subscriber"""
    def __init__(self):
        self._odberatele = {}

    def prihlasit(self, udalost, callback):
        if udalost not in self._odberatele:
            self._odberatele[udalost] = []
        self._odberatele[udalost].append(callback)

    def odhlasit(self, udalost, callback):
        if udalost in self._odberatele:
            self._odberatele[udalost].remove(callback)

    def vyvolat(self, udalost, data=None):
        for callback in self._odberatele.get(udalost, []):
            callback(data)

# Pouziti
events = EventSystem()
events.prihlasit("login", lambda u: print(f"  LOG: Uzivatel {u} se prihlasil"))
events.prihlasit("login", lambda u: print(f"  EMAIL: Vitejte, {u}!"))
events.vyvolat("login", "Alice")
Strategy — definuje rodinu algoritmů a umožňuje je zaměňovat za běhu. Třída Razeni přijímá strategii jako funkci.
def bubble_sort(data):
    arr = data.copy()
    for i in range(len(arr)):
        for j in range(len(arr)-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

def quick_sort(data):
    if len(data) <= 1: return data
    pivot = data[len(data)//2]
    return (quick_sort([x for x in data if x < pivot])
          + [x for x in data if x == pivot]
          + quick_sort([x for x in data if x > pivot]))

class Razeni:
    """Strategy - zamenitelne algoritmy"""
    def __init__(self, strategie=None):
        self.strategie = strategie or sorted

    def serad(self, data):
        return self.strategie(data)

data = [5, 2, 8, 1, 9]
for nazev, fn in [("Bubble", bubble_sort), ("Quick", quick_sort), ("Built-in", sorted)]:
    r = Razeni(fn)
    print(f"  {nazev}: {r.serad(data)}")
Decorator — dynamicky přidává funkcionalitu objektu. Zde: MlekoDekorator a CukrDekorator obalují základní kávu.
class Kava:
    def cena(self): return 50
    def popis(self): return "Kava"

class MlekoDekorator:
    def __init__(self, kava):
        self._kava = kava
    def cena(self): return self._kava.cena() + 15
    def popis(self): return self._kava.popis() + " + mleko"

class CukrDekorator:
    def __init__(self, kava):
        self._kava = kava
    def cena(self): return self._kava.cena() + 5
    def popis(self): return self._kava.popis() + " + cukr"

# Obalujeme zakladni kavu dekoratory
kava = Kava()
kava = MlekoDekorator(kava)
kava = CukrDekorator(kava)
print(f"  {kava.popis()}: {kava.cena()} Kc")  # Kava + mleko + cukr: 70 Kc
