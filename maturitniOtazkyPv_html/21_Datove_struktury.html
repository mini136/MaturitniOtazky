Stack — kontrola závorek. Zásobník (LIFO) pro párování otevíracích a zavíracích závorek. Pokud na konci není stack prázdný, závorky nejsou správně spárované.
class Stack:
    """Zasobnik (LIFO)"""
    def __init__(self):
        self._data = []

    def push(self, item):
        self._data.append(item)

    def pop(self):
        if self.je_prazdny():
            raise IndexError("Stack je prazdny!")
        return self._data.pop()

    def peek(self):
        if self.je_prazdny():
            raise IndexError("Stack je prazdny!")
        return self._data[-1]

    def je_prazdny(self):
        return len(self._data) == 0

    def __len__(self):
        return len(self._data)

def kontrola_zavorek(vyraz):
    """Kontrola sparovani zavorek pomoci stacku"""
    stack = Stack()
    pary = {')': '(', ']': '[', '}': '{'}

    for znak in vyraz:
        if znak in '([{':
            stack.push(znak)
        elif znak in ')]}':
            if stack.je_prazdny() or stack.pop() != pary[znak]:
                return False
    return stack.je_prazdny()

testy = ["((1+2)*3)", "([{}])", "(()", "{[}]", ""]
for vyraz in testy:
    print(f"  '{vyraz}': {'OK' if kontrola_zavorek(vyraz) else 'CHYBA'}")
Queue — simulace tiskárny. Fronta (FIFO) zpracovává tiskové úlohy v pořadí, v jakém přišly.
from collections import deque

class Queue:
    """Fronta (FIFO)"""
    def __init__(self):
        self._data = deque()

    def enqueue(self, item):
        self._data.append(item)

    def dequeue(self):
        if self.je_prazdna():
            raise IndexError("Queue je prazdna!")
        return self._data.popleft()

    def je_prazdna(self):
        return len(self._data) == 0

class Tiskarna:
    def __init__(self):
        self.fronta = Queue()

    def pridej_ulohu(self, dokument):
        self.fronta.enqueue(dokument)
        print(f"  Pridano: {dokument}")

    def tiskni(self):
        while not self.fronta.je_prazdna():
            dok = self.fronta.dequeue()
            print(f"  Tisknu: {dok}")

tiskarna = Tiskarna()
for doc in ["zprava.pdf", "faktura.pdf", "smlouva.doc"]:
    tiskarna.pridej_ulohu(doc)
tiskarna.tiskni()
Linked List — spojový seznam s operacemi pridej, vloz, smaz. Každý uzel (Node) obsahuje data a odkaz na další uzel.
class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def pridej(self, data):
        novy = Node(data)
        novy.next = self.head
        self.head = novy

    def vloz_na_konec(self, data):
        novy = Node(data)
        if not self.head:
            self.head = novy
            return
        aktualni = self.head
        while aktualni.next:
            aktualni = aktualni.next
        aktualni.next = novy

    def smaz(self, data):
        if not self.head: return
        if self.head.data == data:
            self.head = self.head.next
            return
        aktualni = self.head
        while aktualni.next:
            if aktualni.next.data == data:
                aktualni.next = aktualni.next.next
                return
            aktualni = aktualni.next

    def __str__(self):
        prvky = []
        aktualni = self.head
        while aktualni:
            prvky.append(str(aktualni.data))
            aktualni = aktualni.next
        return " -> ".join(prvky) + " -> None"

ll = LinkedList()
for x in [3, 2, 1]:
    ll.pridej(x)
ll.vloz_na_konec(4)
print(f"  {ll}")      # 1 -> 2 -> 3 -> 4 -> None
ll.smaz(2)
print(f"  {ll}")      # 1 -> 3 -> 4 -> None
BST (Binary Search Tree) — binární vyhledávací strom. Levý podstrom < kořen < pravý podstrom. Metoda inorder() vrací seřazené prvky.
class BSTNode:
    def __init__(self, klic):
        self.klic = klic
        self.levy = None
        self.pravy = None

class BST:
    def __init__(self):
        self.koren = None

    def vloz(self, klic):
        self.koren = self._vloz(self.koren, klic)

    def _vloz(self, uzel, klic):
        if uzel is None:
            return BSTNode(klic)
        if klic < uzel.klic:
            uzel.levy = self._vloz(uzel.levy, klic)
        elif klic > uzel.klic:
            uzel.pravy = self._vloz(uzel.pravy, klic)
        return uzel

    def hledej(self, klic):
        return self._hledej(self.koren, klic)

    def _hledej(self, uzel, klic):
        if uzel is None: return False
        if klic == uzel.klic: return True
        if klic < uzel.klic: return self._hledej(uzel.levy, klic)
        return self._hledej(uzel.pravy, klic)

    def inorder(self):
        vysledek = []
        self._inorder(self.koren, vysledek)
        return vysledek

    def _inorder(self, uzel, vysledek):
        if uzel:
            self._inorder(uzel.levy, vysledek)
            vysledek.append(uzel.klic)
            self._inorder(uzel.pravy, vysledek)

bst = BST()
for x in [5, 3, 7, 1, 4, 6, 8]:
    bst.vloz(x)
print(f"  Inorder: {bst.inorder()}")     # [1, 3, 4, 5, 6, 7, 8]
print(f"  Hledam 4: {bst.hledej(4)}")    # True
print(f"  Hledam 9: {bst.hledej(9)}")    # False
MinHeap — binární halda s minimem na vrcholu. push() a pop() mají složitost O(log n). Používá se jako prioritní fronta.
class MinHeap:
    def __init__(self):
        self.data = []

    def push(self, val):
        self.data.append(val)
        self._bubble_up(len(self.data) - 1)

    def pop(self):
        if not self.data:
            raise IndexError("Heap prazdny!")
        min_val = self.data[0]
        posledni = self.data.pop()
        if self.data:
            self.data[0] = posledni
            self._sift_down(0)
        return min_val

    def _bubble_up(self, i):
        while i > 0:
            rodic = (i - 1) // 2
            if self.data[i] < self.data[rodic]:
                self.data[i], self.data[rodic] = self.data[rodic], self.data[i]
                i = rodic
            else:
                break

    def _sift_down(self, i):
        n = len(self.data)
        while True:
            nejmensi = i
            levy = 2*i + 1
            pravy = 2*i + 2
            if levy < n and self.data[levy] < self.data[nejmensi]:
                nejmensi = levy
            if pravy < n and self.data[pravy] < self.data[nejmensi]:
                nejmensi = pravy
            if nejmensi != i:
                self.data[i], self.data[nejmensi] = self.data[nejmensi], self.data[i]
                i = nejmensi
            else:
                break

heap = MinHeap()
for x in [5, 3, 7, 1, 4]:
    heap.push(x)

serazeno = []
while heap.data:
    serazeno.append(heap.pop())
print(f"  Heap sort: {serazeno}")  # [1, 3, 4, 5, 7]
Hash tabulka — implementace s řetězením (chaining) pro řešení kolizí. Každý bucket je seznam párů [klíč, hodnota].
class HashTabulka:
    """Hash tabulka s chainingem"""

    def __init__(self, velikost=10):
        self.velikost = velikost
        self.tabulka = [[] for _ in range(velikost)]

    def _hash(self, klic):
        return hash(klic) % self.velikost

    def vloz(self, klic, hodnota):
        idx = self._hash(klic)
        for par in self.tabulka[idx]:
            if par[0] == klic:
                par[1] = hodnota
                return
        self.tabulka[idx].append([klic, hodnota])

    def najdi(self, klic):
        idx = self._hash(klic)
        for par in self.tabulka[idx]:
            if par[0] == klic:
                return par[1]
        raise KeyError(f"Klic '{klic}' nenalezen")

    def smaz(self, klic):
        idx = self._hash(klic)
        for i, par in enumerate(self.tabulka[idx]):
            if par[0] == klic:
                del self.tabulka[idx][i]
                return
        raise KeyError(f"Klic '{klic}' nenalezen")

ht = HashTabulka()
ht.vloz("alice", 95)
ht.vloz("bob", 87)
ht.vloz("charlie", 92)
print(f"  alice: {ht.najdi('alice')}")    # 95
ht.smaz("bob")
print(f"  Tabulka: {[b for b in ht.tabulka if b]}")
