Ukázkový kód — Java

Stack — kontrola závorek. V Javě používáme Deque<Character> jako zásobník (LIFO). ArrayDeque je rychlejší než starý Stack. Závorky párujeme při prohledávání vstupu.
import java.util.ArrayDeque;
import java.util.Deque;

public class KontrolaZavorek {
    public static boolean kontroluj(String vyraz) {
        Deque<Character> stack = new ArrayDeque<>();

        for (char c : vyraz.toCharArray()) {
            if (c == '(' || c == '[' || c == '{') {
                stack.push(c);
            } else if (c == ')' || c == ']' || c == '}') {
                if (stack.isEmpty()) return false;
                char top = stack.pop();
                if ((c == ')' && top != '(') ||
                    (c == ']' && top != '[') ||
                    (c == '}' && top != '{')) return false;
            }
        }
        return stack.isEmpty();
    }

    public static void main(String[] args) {
        String[] testy = {"((1+2)*3)", "([{}])", "(()", "{[}]"};
        for (String v : testy) {
            System.out.println("'" + v + "': " + (kontroluj(v) ? "OK" : "CHYBA"));
        }
        // '((1+2)*3)': OK
        // '([{}])':    OK
        // '(()':       CHYBA
        // '{[}]':      CHYBA
    }
}
Queue — simulace tiskárny. Rozhraní Queue<E> s implementací LinkedList. offer() přidá na konec, poll() odebere ze začátku (FIFO).
import java.util.LinkedList;
import java.util.Queue;

public class Tiskarna {
    private Queue<String> fronta = new LinkedList<>();

    public void pridejUlohu(String dokument) {
        fronta.offer(dokument);
        System.out.println("Pridano: " + dokument);
    }

    public void tiskni() {
        while (!fronta.isEmpty()) {
            System.out.println("Tisknu: " + fronta.poll());
        }
    }

    public static void main(String[] args) {
        Tiskarna t = new Tiskarna();
        t.pridejUlohu("zprava.pdf");    // Pridano: zprava.pdf
        t.pridejUlohu("faktura.pdf");
        t.pridejUlohu("smlouva.doc");
        t.tiskni();
        // Tisknu: zprava.pdf   (FIFO poradi)
        // Tisknu: faktura.pdf
        // Tisknu: smlouva.doc
    }
}
Linked List — generický spojový seznam. Každý uzel Node<T> drží data a odkaz na další uzel. Java má vestavěný java.util.LinkedList, zde ukázka vlastní implementace.
public class SpojovySeznam<T> {
    private static class Node<T> {
        T data;
        Node<T> next;
        Node(T data) { this.data = data; }
    }

    private Node<T> head;

    public void pridejNaZacatek(T data) {
        Node<T> novy = new Node<>(data);
        novy.next = head;
        head = novy;
    }

    public void pridejNaKonec(T data) {
        Node<T> novy = new Node<>(data);
        if (head == null) { head = novy; return; }
        Node<T> aktualni = head;
        while (aktualni.next != null) aktualni = aktualni.next;
        aktualni.next = novy;
    }

    public void smaz(T data) {
        if (head == null) return;
        if (head.data.equals(data)) { head = head.next; return; }
        Node<T> aktualni = head;
        while (aktualni.next != null) {
            if (aktualni.next.data.equals(data)) {
                aktualni.next = aktualni.next.next;
                return;
            }
            aktualni = aktualni.next;
        }
    }

    @Override
    public String toString() {
        StringBuilder sb = new StringBuilder();
        Node<T> aktualni = head;
        while (aktualni != null) {
            sb.append(aktualni.data).append(" -> ");
            aktualni = aktualni.next;
        }
        return sb + "null";
    }

    public static void main(String[] args) {
        SpojovySeznam<Integer> ll = new SpojovySeznam<>();
        ll.pridejNaZacatek(2);
        ll.pridejNaZacatek(1);    // 1 -> 2 -> null
        ll.pridejNaKonec(3);      // 1 -> 2 -> 3 -> null
        System.out.println(ll);
        ll.smaz(2);
        System.out.println(ll);   // 1 -> 3 -> null
    }
}
BST (Binary Search Tree) — binární vyhledávací strom v Javě. Levý podstrom < kořen < pravý podstrom. In-order průchod vrátí prvky seřazené vzestupně.
public class BST {
    private static class Node {
        int klic;
        Node levy, pravy;
        Node(int klic) { this.klic = klic; }
    }

    private Node koren;

    public void vloz(int klic) { koren = vloz(koren, klic); }

    private Node vloz(Node uzel, int klic) {
        if (uzel == null) return new Node(klic);
        if (klic < uzel.klic)      uzel.levy  = vloz(uzel.levy,  klic);
        else if (klic > uzel.klic) uzel.pravy = vloz(uzel.pravy, klic);
        return uzel;
    }

    public boolean hledej(int klic) { return hledej(koren, klic); }

    private boolean hledej(Node uzel, int klic) {
        if (uzel == null) return false;
        if (klic == uzel.klic) return true;
        return klic < uzel.klic
            ? hledej(uzel.levy,  klic)
            : hledej(uzel.pravy, klic);
    }

    public void inorder(Node uzel) {
        if (uzel == null) return;
        inorder(uzel.levy);
        System.out.print(uzel.klic + " ");
        inorder(uzel.pravy);
    }

    public static void main(String[] args) {
        BST bst = new BST();
        for (int x : new int[]{5, 3, 7, 1, 4, 6, 8}) bst.vloz(x);
        bst.inorder(bst.koren);   // 1 3 4 5 6 7 8  (serazene!)
        System.out.println();
        System.out.println(bst.hledej(4));  // true
        System.out.println(bst.hledej(9));  // false
    }
}
PriorityQueue (MinHeap) — Java PriorityQueue je MinHeap. offer() vloží v O(log n), poll() odebere minimum v O(log n), peek() přečte minimum v O(1).
import java.util.PriorityQueue;
import java.util.Comparator;

public class HeapUkazka {
    public static void main(String[] args) {
        // MinHeap (vychozi chovani PriorityQueue)
        PriorityQueue<Integer> minHeap = new PriorityQueue<>();
        for (int x : new int[]{5, 3, 7, 1, 4}) minHeap.offer(x);

        System.out.print("HeapSort (min): ");
        while (!minHeap.isEmpty()) {
            System.out.print(minHeap.poll() + " ");  // 1 3 4 5 7
        }

        // MaxHeap - obraceny komparator
        PriorityQueue<Integer> maxHeap = new PriorityQueue<>(Comparator.reverseOrder());
        maxHeap.offer(5); maxHeap.offer(3); maxHeap.offer(7); maxHeap.offer(1);
        System.out.println("\nMax: " + maxHeap.peek());   // 7
        System.out.println("Poll: " + maxHeap.poll());    // 7 odebrano
    }
}
HashMap, HashSet a TreeSet — tři nejdůležitější kolekce pro práci s klíči/hodnotami. TreeSet je interně Red-Black strom, zajišťuje O(log n) a seřazené pořadí při iteraci.
import java.util.*;

public class KolekceUkazka {
    public static void main(String[] args) {
        // HashMap: klic -> hodnota, O(1) prumer, bez poradi
        Map<String, Integer> scores = new HashMap<>();
        scores.put("Alice", 95);
        scores.put("Bob",   87);
        scores.put("Alice", 99);          // prepis existujiciho klice
        System.out.println(scores.get("Alice"));          // 99
        System.out.println(scores.containsKey("Bob"));    // true
        scores.forEach((k, v) -> System.out.println(k + ": " + v));

        // HashSet: pouze unikatni prvky, O(1), bez poradi
        Set<String> hashSet = new HashSet<>(Arrays.asList("C", "A", "B", "A"));
        System.out.println(hashSet);    // [A, B, C] - poporadi neni zaruceno

        // TreeSet: unikatni + serazene, O(log n), Red-Black strom
        TreeSet<Integer> treeSet = new TreeSet<>();
        treeSet.add(5); treeSet.add(1); treeSet.add(3);
        treeSet.add(2); treeSet.add(1); // duplikat ignorovan
        System.out.println(treeSet);            // [1, 2, 3, 5] - serazene!
        System.out.println(treeSet.first());    // 1  - minimum O(log n)
        System.out.println(treeSet.last());     // 5  - maximum O(log n)
        System.out.println(treeSet.floor(4));   // 3  - nejvetsi <= 4
        System.out.println(treeSet.ceiling(2)); // 2  - nejmensi >= 2
        System.out.println(treeSet.headSet(3)); // [1, 2]     - prvky < 3
        System.out.println(treeSet.tailSet(3)); // [3, 5]     - prvky >= 3
    }
}
