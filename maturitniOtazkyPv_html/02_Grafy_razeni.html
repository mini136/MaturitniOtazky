Ukázkový kód — Python (BFS, DFS, Dijkstra)

Reprezentace grafu seznamem sousednosti a průchod BFS (do šířky) pomocí fronty deque. Vrací seznam navštívených uzlů v pořadí procházení.
from collections import deque

# Reprezentace grafu - seznam sousednosti
graf = {
    'A': ['B', 'C'],
    'B': ['A', 'D', 'E'],
    'C': ['A', 'F'],
    'D': ['B'],
    'E': ['B', 'F'],
    'F': ['C', 'E']
}

def bfs(graf, start):
    """Pruchod do sirky (Breadth-First Search)"""
    navstivene = set()
    fronta = deque([start])
    navstivene.add(start)
    poradi = []

    while fronta:
        uzel = fronta.popleft()  # FIFO
        poradi.append(uzel)
        for soused in graf[uzel]:
            if soused not in navstivene:
                navstivene.add(soused)
                fronta.append(soused)
    return poradi

print("BFS:", bfs(graf, 'A'))  # ['A', 'B', 'C', 'D', 'E', 'F']
DFS (do hloubky) — rekurzivní verze. Prochází co nejdále jednou větví, pak se vrací (backtracking).
def dfs(graf, start, navstivene=None):
    """Pruchod do hloubky (Depth-First Search)"""
    if navstivene is None:
        navstivene = set()
    navstivene.add(start)
    poradi = [start]
    for soused in graf[start]:
        if soused not in navstivene:
            poradi.extend(dfs(graf, soused, navstivene))
    return poradi

print("DFS:", dfs(graf, 'A'))  # ['A', 'B', 'D', 'E', 'F', 'C']
Dijkstrův algoritmus — nalezení nejkratší cesty v ohodnoceném grafu s nezápornými vahami. Používá prioritní frontu (heapq).
import heapq

vahovany_graf = {
    'A': [('B', 4), ('C', 2)],
    'B': [('D', 3), ('C', 1)],
    'C': [('B', 1), ('D', 5)],
    'D': []
}

def dijkstra(graf, start):
    """Nejkratsi cesty z pocatecniho uzlu"""
    vzdalenosti = {uzel: float('inf') for uzel in graf}
    vzdalenosti[start] = 0
    prioritni_fronta = [(0, start)]

    while prioritni_fronta:
        aktualni_vzdalenost, aktualni = heapq.heappop(prioritni_fronta)
        if aktualni_vzdalenost > vzdalenosti[aktualni]:
            continue
        for soused, vaha in graf[aktualni]:
            vzdalenost = aktualni_vzdalenost + vaha
            if vzdalenost < vzdalenosti[soused]:
                vzdalenosti[soused] = vzdalenost
                heapq.heappush(prioritni_fronta, (vzdalenost, soused))
    return vzdalenosti

print("Dijkstra:", dijkstra(vahovany_graf, 'A'))

Ukázkový kód — Řadicí algoritmy

Bubble Sort — opakovaně porovnává sousední prvky a prohazuje je. Optimalizace s příznakem prohozeno pro včasné ukončení.
def bubble_sort(pole):
    """O(n²) - porovnava sousedni prvky, stabilni"""
    n = len(pole)
    for i in range(n):
        prohozeno = False
        for j in range(0, n - i - 1):
            if pole[j] > pole[j + 1]:
                pole[j], pole[j + 1] = pole[j + 1], pole[j]
                prohozeno = True
        if not prohozeno:
            break
    return pole
Merge Sort — rozděl a panuj. Rekurzivně dělí pole na poloviny a poté slučuje seřazené části. Stabilní, O(n log n).
def merge_sort(pole):
    """O(n log n) - rozdel a panuj, stabilni"""
    if len(pole) <= 1:
        return pole
    stred = len(pole) // 2
    leva = merge_sort(pole[:stred])
    prava = merge_sort(pole[stred:])

    vysledek = []
    i = j = 0
    while i < len(leva) and j < len(prava):
        if leva[i] <= prava[j]:
            vysledek.append(leva[i]); i += 1
        else:
            vysledek.append(prava[j]); j += 1
    vysledek.extend(leva[i:])
    vysledek.extend(prava[j:])
    return vysledek
Quick Sort — volí pivot a rozděluje pole na menší a větší prvky. Průměrně O(n log n), ale worst-case O(n²).
def quick_sort(pole):
    """O(n log n) prumerne - pivot, nestabilni"""
    if len(pole) <= 1:
        return pole
    pivot = pole[len(pole) // 2]
    mensi = [x for x in pole if x < pivot]
    stejne = [x for x in pole if x == pivot]
    vetsi = [x for x in pole if x > pivot]
    return quick_sort(mensi) + stejne + quick_sort(vetsi)
BFS aplikace — hledání cesty v bludišti. Mřížka s překážkami (#), hledá nejkratší cestu z S do E.
def bfs_bludiste(bludiste, start, cil):
    """Najde nejkratsi cestu v bludisti pomoci BFS"""
    radky, sloupce = len(bludiste), len(bludiste[0])
    fronta = deque([(start, [start])])
    navstivene = {start}

    while fronta:
        (r, c), cesta = fronta.popleft()
        if (r, c) == cil:
            return cesta

        for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
            nr, nc = r + dr, c + dc
            if (0 <= nr < radky and 0 <= nc < sloupce
                and bludiste[nr][nc] != '#'
                and (nr, nc) not in navstivene):
                navstivene.add((nr, nc))
                fronta.append(((nr, nc), cesta + [(nr, nc)]))
    return None

bludiste = [
    ['S', '.', '.', '#', '.'],
    ['#', '#', '.', '#', '.'],
    ['.', '.', '.', '.', '.'],
    ['.', '#', '#', '#', '.'],
    ['.', '.', '.', '.', 'E']
]
cesta = bfs_bludiste(bludiste, (0,0), (4,4))
print(f"Cesta bludistem: {cesta}")
